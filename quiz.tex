\begin{question}[class=medium]\addpoints*{3}
Given two functions: $w(x)=x^{\vary{2}{3}{4}}$ and $h(x)=\vary{2}{3}{4}x^{\vary{}{2}{3}}$
Write the equation of a line with slope $h(x)$ that intersects $y=w(x)$ at the point where $x=2$.
\end{question}
\begin{solution}
$y-\vary{4}{8}{16}=\vary{4}{12}{32}(x-2)$
\end{solution}
